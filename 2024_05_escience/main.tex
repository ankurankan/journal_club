					\item Edges are interpreted as causal direction.

\begin{frame}{State of Simulation Features in pgmpy}

	\begin{itemize}

		\item Certain/Uncertain Intervention.
		\item Time-Series data.
		\item Partial simulations.
	\end{itemize}
